	상태나 구조에 의해서 선택이 이루어진다.
	상태 : 	:link	: visited	:hover	:active
	
	a:link{
		color : red;
		text-decoration: none; 
	}
	a:visited{
		color : green;
	}
	a:hover{
		color : blue;
		text-decoration: underline;
	}
	a:active{
		color : black;
		font-size: 1.2em;
	}
	p:hover{	/* p태그는 링크가 아니기때문에 link가 없다 */
		background: lime;
	}

	구조에 의한 의사코드
	td:nth-child(3) - 3번째 td를 선택
	td:nth=child(even) - 짝수번째 td를 선택
	td:nth-child(odd) - 홀수번째 td를 선택
	tr:nth-child(2n) - 짝수번째 tr을 선택
	
	#tab1{
		border : 2px solid blue;
		border-collapse: collapse;
	}
	#tab1 td{
		width : 150px;
		height : 40px;
	}
	#tab1 .t1{
		background: lightgray;
	}
	
	#tab2{
		border : 5px inset green;
		border-spacing: 10px;
	}
	#tab2 td{
		width : 150px;
		height : 50px;
	}
	#tab2 tr:nth-child(even){
		background: lightgreen;
	}
	
	#tab3 td{
		width : 120px;
		height : 30px;
	}
	
	#tab3{
		border : 4px double red;
	}
	
	#tab3 td:nth-child(odd){ /* odd 대신 2n-1로 적어도 된다. */
		background: lightblue;
	}
