속성선택자
	
	p{
		font-size: 1.5em;
	}
	
	p:hover{
		color : red;
	}
	
	p:active{
		background: pink;
	}
	
	p[title]{ /* title을 가진 p만 선택 (첫번째, 두번째)*/
		background: crimson;
			
	}
	
	p[title = 무궁화]{ /* 무궁화 title을 가진 p만 선택 (첫번째)*/
		color : white;
	}
	/* p.pp */
	p[class = pp]{ /* (첫번째, 세번째) */
		font-weight: bold;
		font-size: 2.0em;
	}
